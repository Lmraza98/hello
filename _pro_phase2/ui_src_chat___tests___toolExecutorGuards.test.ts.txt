==================== ui/src/chat/__tests__/toolExecutorGuards.test.ts ====================
import { describe, expect, test, vi, beforeEach } from 'vitest';

import { resolveTemplate } from '../toolExecutor/template';

const executeToolMock = vi.fn();

vi.mock('../toolExecutor/config', () => ({
  TOOL_LANE_SERIAL: false,
  TOOL_LANE_KEY: 'chat-tools',
}));

vi.mock('../toolExecutor/executeTool', () => ({
  executeTool: (...args: unknown[]) => executeToolMock(...args),
}));

import { dispatchToolCalls } from '../toolExecutor/dispatch';

describe('toolExecutor guards', () => {
  beforeEach(() => {
    executeToolMock.mockReset();
  });

  test('resolveTemplate supports $prev and $tool.<name>', () => {
    const previous = { ref: 'e123', items: [{ id: 7 }] };
    const byTool = {
      search_companies: [{ company_id: 42, name: 'Acme' }],
    };

    const resolved = resolveTemplate(
      {
        ref: '$prev.ref',
        itemId: '$prev.items.0.id',
        companyId: '$tool.search_companies.0.company_id',
      },
      previous,
      byTool
    ) as Record<string, unknown>;

    expect(resolved.ref).toBe('e123');
    expect(resolved.itemId).toBe(7);
    expect(resolved.companyId).toBe(42);
  });

  test('dispatchToolCalls executes tools in order', async () => {
    executeToolMock.mockResolvedValue({ ok: true });
    const order: string[] = [];

    const result = await dispatchToolCalls(
      [
        { name: 'browser_health', args: {} },
        { name: 'browser_tabs', args: {} },
      ],
      (name) => order.push(name)
    );

    expect(order).toEqual(['browser_health', 'browser_tabs']);
    expect(result.executed.map((x) => x.name)).toEqual(['browser_health', 'browser_tabs']);
  });

  test('stop-chain-on-failure behavior preserved for browser tools', async () => {
    executeToolMock
      .mockResolvedValueOnce({ error: true, message: 'boom' })
      .mockResolvedValue({ ok: true });

    const result = await dispatchToolCalls([
      { name: 'browser_health', args: {} },
      { name: 'browser_tabs', args: {} },
    ]);

    expect(result.executed).toHaveLength(1);
    expect(result.executed[0]?.name).toBe('browser_health');
  });
});
